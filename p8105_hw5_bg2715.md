``` r
tidy_df = p2_df  %>% 
  separate(file_name, into = c('arm','ID'),sep = '_') %>%
  mutate(ID = as.integer(ID)) %>%
  pivot_longer(week_1:week_8, 
               names_to = "week",
               values_to = "value",
               names_prefix = 'week_') %>%
  mutate(week = as.factor(week)) %>%
  relocate(ID)
```
